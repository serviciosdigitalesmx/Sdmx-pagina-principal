    14	function decodeJwtPayload(token: string) {
    15	  try {
    16	    const parts = token.split(".");
    17	    if (parts.length !== 3) {
    18	      return null;
    19	    }
    20	
    21	    const payload = parts[1];
    22	    const decoded = atob(payload.replace(/-/g, "+").replace(/_/g, "/"));
    23	    return JSON.parse(decoded) as Record<string, unknown>;
    24	  } catch {
    25	    return null;
    26	  }
    27	}
    28	
    29	export function getCurrentSession(): CurrentSession | null {
    30	  if (typeof window === "undefined") {
    31	    return null;
    32	  }
    33	
    34	  const token = readAuthToken();
    35	
    36	  if (!token) {
    37	    return null;
    38	  }
    39	
    40	  const decoded = decodeJwtPayload(token);
    41	
    42	  if (!decoded) {
    43	    return null;
    44	  }
    45	
    46	  const tenantSlug = typeof decoded.tenant_slug === "string" ? decoded.tenant_slug.trim() : "";
    47	  const tenantId = typeof decoded.tenant_id === "string" ? decoded.tenant_id.trim() : "";
    48	  const email = typeof decoded.email === "string" ? decoded.email.trim() : "";
    49	  const role = typeof decoded.role === "string" ? decoded.role.trim() : "";
    50	  const sucursalId = typeof decoded.sucursal_id === "string" ? decoded.sucursal_id.trim() : "";
    51	
    52	  if (!tenantSlug) {
    53	    return null;
    54	  }
    55	
    56	  return {
    57	    token,
    58	    email,
    59	    role,
    60	    tenantId,

17) Diagnóstico automático
============================================================
Revisar arriba:
- Si POST /api/auth/exchange con token inválido devuelve 401, el endpoint está vivo.
- Si devuelve 500 con token inválido, el controlador/middleware truena antes de validar token.
- Si exchange real ya da 200 pero dashboard falla, el culpable es una petición posterior.
- Si CORS no devuelve Access-Control-Allow-Credentials cuando usas cookies, revisar CORS.
- Si cookie no llega en requests posteriores, revisar SameSite/Secure/Domain/credentials: include.
- Si una ruta tenant devuelve 500 sin cookie, revisar middleware auth/tenant.
